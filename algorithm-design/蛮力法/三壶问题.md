vector<int*> new_state(int state[3]){//vector<int[3]>是不允许的
    vector<int*> res;
    for (int i = 0; i <3 ; ++i) {
        if(state[i]>0) {
            for (int j = 0; j < 3; ++j) {
                if (j != i) {
                    int* tmp=new int[3];//如果在这里使用int tmp[3],这种在栈区的使用方法，函数结束后指向的内存会被释放，因而会res里的结果会是乱码
                    tmp[0]=state[0];//tmp[3]放在外面更不对
                    tmp[1]=state[1];
                    tmp[2]=state[2];
                    if(full[j]-state[j]>state[i]) {
                       tmp[j]+=state[i];
                       tmp[i]=0;
                   } else{
                       tmp[j]=full[j];
                       tmp[i]-=full[j]-state[j];
                   }
                    res.push_back(tmp);
                }
            }
        }
    }
    return res;
}
