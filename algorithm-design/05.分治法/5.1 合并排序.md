```cpp
//书上伪代码做了copy，实际完全可以不用做copy,用下标即可
void belt_merge(int low,int high,int mid){
    int left=low;
    int right=mid+1;
    int ptr=left;
    while(left<=mid && right<=high){
        if (input[left]<input[right]){
            help[ptr++]=input[left++];
        }else{
            help[ptr++]=input[right++];
        }
    }
    while(left<=mid){
        help[ptr++]=input[left++];
    }
    while(right<=high){
        help[ptr++]=input[right++];
    }
    for (int i = low; i <=high ; ++i) {
        input[i]=help[i];
    }
}
void merge_sort(int low,int high){
    if(low<high){
        int mid=(low+high)/2;
        merge_sort(low,mid);
        merge_sort(mid+1,high);
        belt_merge(low,high,mid);
    }

}
int main(){
    input={77,88,7,99,7,32,17};
    help=input;
    merge_sort(0,input.size()-1);
    for(auto pp:input){
        cout<<pp<<endl;
    }
    return 0;
```
注意合并排序本身是稳定的；合并排序的效率有两个角度去计算，一个是比较次数，一个是键值的移动次数
